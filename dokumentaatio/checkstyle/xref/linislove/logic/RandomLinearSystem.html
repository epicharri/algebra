6   package linislove.logic;
7   
8   import linislove.mylittlemath.LinearSystem;
9   import linislove.mylittlemath.Rational;
10  
11  /**
12   * Luokka satunnaisen kvadraattisen yhtälöryhmän luomista varten.
13   *
14   * @author harrikah
15   */
16  public class RandomLinearSystem {
17  
18      /**
19       * Luo satunnaisen kvadraattisen yhtälöryhmän.
20       *
21       * @param x Tuntemattomien ja samalla yhtälöiden määrä.
22       * @param maxNum Suurin arvo osoittajalle
23       * @param maxDenom Suurin arvo nimittäjälle
24       * @return LinearSystem -luokan olio.
25       * @throws java.lang.Exception jos rationaalilukua luodessa tulee ongelma.
26       *
27       */
28      public static LinearSystem create(int x, int maxNum, int maxDenom) throws Exception {
29          StringBuilder str = new StringBuilder();
30          for (int i = 0; i < x; i++) {
31              for (int j = 0; j < x; j++) {
32                  int subscript = j + 1;
33                  str.
34                          append(randomPlusOrMinus()).
35                          append(randomRational(maxNum, maxDenom).toString()).
36                          append('x').
37                          append(subscript);
38              }
39              str.append('=').append(randomPlusOrMinus()).
40                      append(randomRational(maxNum, maxDenom).toString()).
41                      append(';');
42          }
43          return new LinearSystem(str.toString());
44      }
45  
46      private static Rational randomRational(int maxNum, int maxDenom) {
47          maxNum = randomInt(maxNum);
48          maxDenom = randomInt(maxDenom) + 1;
49          return new Rational(maxNum, (maxDenom));
50      }
51  
52      private static int randomInt(int maxInt) {
53          return (int) (Math.random() * maxInt);
54      }
55  
56      private static String randomPlusOrMinus() {
57          return (Math.random() >= 0.5) ? "-" : "+";
58      }
59  
60  }
