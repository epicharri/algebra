16  /**
17   * Luokka sisältää ratkaisumetodit vektorijonon vapauden selvittämiseen ja
18   * yhtälöryhmän ratkaisuun.
19   *
20   * @author harrikah
21   */
22  public class Solution {
23  
24      /**
25       * Metodi ratkaisee kvadraattisen yhtälöryhmän.
26       *
27       * @param system Merkkijono yhtälöryhmästä.
28       * @return Yhtälöryhmän ratkaisu merkkijonona.
29       * @throws java.lang.Exception jos syöte on ollut virheellinen.
30       *
31       */
32      public static String solveLinearSystem(String system) throws Exception {
33          if (lolCat(system)) {
34              return lolCatAnswer("yhtälöryhmä");
35          }
36          LinearSystem linearSystem = new LinearSystem(system);
37          Rational[] x = LinearSystemSolver.solveSystem(linearSystem);
38          return Count.giveSolutions(x);
39      }
40  
41      /**
42       * Metodi ratkaisee vektorijonon vapauden ja ilmoittaa jos se on myös kanta.
43       *
44       * @param vectorSet Merkkijono vektorijonosta.
45       * @return Ratkaisu merkkijonona.
46       * @throws java.lang.Exception jos syöte on virheellinen.
47       *
48       */
49      public static String solveLinearDependency(String vectorSet) throws Exception {
50          if (lolCat(vectorSet)) {
51              return lolCatAnswer("vektorijono");
52          }
53          try {
54              vectorSet = vectorSet.replaceAll(" ", "");
55              if (vectorSet.matches("\\(\\)")) {
56                  return freeTrivially();
57              }
58              Matrix matrixA = new Matrix(new SetOfVectors(vectorSet));
59              // Lineaarialgebraa: m on rivien määrä ja n on sarakkeiden määrä.
60              // Matriisi on luotu vektoreista ja selvitetään vapautta, joten
61              // m on siten komponenttien määrä ja n on vektoreiden määrä.
62              int m = matrixA.getM();
63              int n = matrixA.getN();
64              if (m < n) {
65                  return moreVectorsThanDimension();
66              }
67              if (m > n) {
68                  return lessVectorsThanDimension(matrixA, m, n);
69              }
70              return numberOfVectorsAndDimensionAreEqual(matrixA);
71          } catch (InputMismatchException e) {
72              throw e;
73          } catch (Exception e) {
74              throw e;
75          }
76      }
77  
78      /**
79       * Metodi tyhjän vektorijonon ratkaisun palauttamista varten.
80       *
81       * @return Merkkijono jossa kerrotaan tyhjän vektorijonon olevan vapaa.
82       */
83      private static String freeTrivially() {
84          return "Vektorijono on vapaa koska se on tyhjä. On sovittu, että tyhjä vektorijono on "
85                  + "lineaarisesti riippumaton.";
86      }
87  
88      /**
89       * Metodi selvittää vektorijonon ratkaisun jos vektorijonossa on yhtä monta
90       * vektoria kuin on vektorissa komponentteja.
91       *
92       * @param matrixA Matrix -olio.
93       * @return Palauttaa merkkijonon jossa kerrotaan ja perustellaan onko
94       * vektorijono vapaa vai sidottu.
95       */
96      private static String numberOfVectorsAndDimensionAreEqual(Matrix matrixA) {
97          Rational det = Count.det(matrixA);
98          String space = "R^" + matrixA.getM();
99          String freedom = "Vektorijonosta muodostetun matriisin "
100                 + "determinantti on " + det + ".\n";
101         if (det.equals(Rational.ZERO)) {
102             freedom += "Koska determinantti on 0, on vektorijono lineaarisesti "
103                     + "riippuvainen eli sidottu. Jono ei myöskään ole vektoriavaruuden "
104                     + space + " kanta, koska vain vapaa jono jossa on "
105                     + matrixA.getM() + " vektoria virittää avaruuden " + space + ".";
106         } else {
107             freedom += "Koska determinantti on nollasta eriävä, on vektorijono "
108                     + "lineaarisesti riippumaton eli vapaa. \n"
109                     + "Koska vektorijono on vapaa ja siinä olevien vektorien määrä "
110                     + "on sama kuin avaruuden " + space + " dimensio, "
111                     + "jono muodostaa avaruuden " + space + " kannan ja "
112                     + "virittää avaruuden " + space + ". \n"
113                     + "Antamillasi vektoreilla siis pääsee ihan joka paikkaan kyseisessä avaruudessa :)";
114         }
115         return freedom;
116     }

118     /**
119      * Metodi ilmoittaa ratkaisun jos vektoreita jonossa on enemmän kuin
120      * avaruuden dimensio.
121      *
122      * @return Merkkijono jossa vastaus että jono on sidottu ja perustelu
123      * siihen.
124      */
125     private static String moreVectorsThanDimension() {
126         return "Vektorijono on lineaarisesti riippuvainen eli "
127                 + "sidottu, koska vektoreiden määrä on suurempi kuin "
128                 + "vektoriavaruuden dimensio.";
129     }
130 
131     /**
132      * Metodi ratkaisee onko vektorijono vapaa vai ei ja palauttaa perustellun
133      * vastauksen koskien vektorijonoa jossa on vähemmän vektoreita kuin
134      * vektorissa on komponentteja.
135      *
136      * @param matrixA Matrix -olio, jossa vektorijono matriisimuodossa.
137      * @param m Matriisin rivien määrä, eli vektorin komponenttien määrä.
138      * @param n Matriisin sarakkeiden määrä eli vektoreiden määrä.
139      * @return
140      */
141     private static String lessVectorsThanDimension(Matrix matrixA, int m, int n) {
142         // Jos m > n niin matriisi A on vapaa joss det(A^T*A) != 0. 
143         Rational detAtA = Count.det(Count.multiply(Count.transpose(matrixA), matrixA));
144         if (detAtA.isZero()) {
145             return "Vektorijono on sidottu.\nVektoreita on "
146                     + "vähemmän kuin vektoriavaruuden dimensio. Tutkitaan siis "
147                     + "vapautta seuraavasti. Olkoon A syöttämästäsi vektorijonosta muodostettu "
148                     + "matriisi siten että vektorien komponentit muodostavat "
149                     + "matriisin sarakkeet. Olkoon m matriisin rivien määrä (ja siis komponenttien määrä) ja "
150                     + "n matriisin sarakkeiden määrä (ja siis vektoreiden määrä). Jos m > n niin A on sidottu "
151                     + "jos ja vain jos det(A^T * A) = 0. Syöttämästäsi vektorijonosta näin muodostetun matriisin "
152                     + "determinantti on 0, joten syöttämäsi vektorijono on lineaarisesti riippuvainen eli sidottu.";
153         } else {
154             return "Vektorijono on vapaa.\n"
155                     + "Vektoreita on vähemmän kuin vektoriavaruuden dimensio. Tutkitaan siis vapautta seuraavasti. "
156                     + "Olkoon A vektorijonosta muodostettu matriisi siten että vektorien komponentit muodostavat "
157                     + "matriisin sarakkeet. Olkoon m matriisin rivien määrä (ja siis komponenttien määrä) ja "
158                     + "n matriisin sarakkeiden määrä (ja siis vektoreiden määrä). Jos m > n niin A on vapaa "
159                     + "jos ja vain jos det(A^T * A) != 0. Syöttämästäsi vektorijonosta näin muodostetun matriisin determinantti on "
160                     + detAtA + ".\n"
161                     + "Koska determinantti on nollasta eriävä, on syöttämäsi vektorijono lineaarisesti riippumaton eli vapaa.";
162         }
163     }
