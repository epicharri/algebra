9       /**
10       * Metodi ratkaisee kvadraattisen yhtälöryhmän.
11       *
12       * @param system LinearSystem -olio joka sisältää yhtälöryhmän tiedot.
13       * @return Ratkaisutaulukko muodossa Rational[] sisältäen ratkaisut siten
14       * että indeksissä 0 on x1:n ratkaisu, indeksissä 1 x2:n ratkaisu jne.
15       * @throws java.lang.Exception jos syötteet virheelliset tai yritetty jakaa
16       * nollalla.
17       *
18       */
19      public static Rational[] solveSystem(LinearSystem system) throws Exception {
20          int rows = system.getB().length;
21          Rational b[] = new Rational[rows];
22          Rational[][] originalB = system.getB();
23          for (int i = 0; i < rows; i++) {
24              b[i] = originalB[i][0];
25          }
26          return solve(Count.createCopy(system.getA()), b);
27      }

29      /**
30       * Metodi ratkaisee yhtälön Ab = x Gaussin eliminointimenetelmällä
31       * osittaispivotoimalla. Huomioitava, että metodi muokkaa parametreina
32       * annettuja taulukoita.
33       *
34       * @param originalA Rational[][] taulukko matriisista A
35       * @param originalB Rational[] taulukko matriisista b
36       * @return Rational[] taulukko jossa yhtälöryhmän vastaukset.
37       *
38       */
39      private static Rational[] solve(Rational originalA[][], Rational originalB[]) throws Exception {
40  
41          int n = originalB.length;
42          int[] index = new int[n];
43          Rational x[] = new Rational[n];
44          Rational a[][] = originalA;
45          Rational b[] = originalB;
46          // Muuttaa matriisin yläkolmiomatriisiksi
47          gaussian(a, index);
48  
49          // Päivitetään b
50          for (int i = 0; i < n - 1; ++i) {
51              for (int j = i + 1; j < n; ++j) {
52                  b[index[j]] = Count.difference(b[index[j]], Count.product(a[index[j]][i], b[index[i]]));
53              }
54          }
55          // Takaisinsijoitus
56          Rational numberInDiagonal = a[index[n - 1]][n - 1];
57  
58          try {
59              x[n - 1] = b[index[n - 1]].divide(numberInDiagonal);
60          } catch (Exception e) {
61              throw new IllegalArgumentException("Yhtälöryhmällä ei ole ratkaisua "
62                      + "tai sillä ei ole yksikäsitteistä ratkaisua.");
63          }
64          for (int i = n - 2; i >= 0; --i) {
65              x[i] = b[index[i]];
66              for (int j = i + 1; j < n; ++j) {
67                  x[i] = x[i].minus(a[index[i]][j].multiply(x[j]));
68              }
69              numberInDiagonal = a[index[i]][i];
70              try {
71                  x[i] = x[i].divide(numberInDiagonal);
72              } catch (Exception e) {
73                  throw new IllegalArgumentException("Yhtälöryhmällä ei ole ratkaisua "
74                          + "tai sillä ei ole yksikäsitteistä ratkaisua.");
75                  /*
76                  if (x[i].isZero() && numberInDiagonal.isZero()) 
77                      throw new IllegalArgumentException("Yhtälöryhmälle ei ole olemassa "
78                          + "yksikäsitteistä ratkaisua.");
79                  else throw new IllegalArgumentException("Yhtälöryhmällä ei ole ratkaisua.");
80                   */
81              }
82          }
83          return x;
84      }

86      /**
87       * Metodi toteuttaa Gaussin eliminointimenetelmällä yläkolmiomatriisin.
88       *
89       * @param a Matriisi (Rational[][] a) joka muunnetaan.
90       * @param index pivotointijärjestys tallennetaan tähän.
91       */
92      public static void gaussian(Rational a[][], int index[]) {
93          //Rational detOfRowOps = Rational.ONE;
94          int n = index.length;
95          Rational c[] = new Rational[n];
96  
97          // Initialisoidaan index
98          for (int i = 0; i < n; ++i) {
99              index[i] = i;
100         }
101 
102         // Etsitään skaalautuvat tekijät, yksi per rivi
103         for (int i = 0; i < n; ++i) {
104             Rational c1 = Rational.ZERO;
105             for (int j = 0; j < n; ++j) {
106                 Rational c0 = Count.abs(a[i][j]);
107                 if (c0.greaterThan(c1)) {
108                     c1 = c0;
109                 }
110             }
111             c[i] = c1;
112         }
113 
114         // Etsitään kunkin sarakkeen pivotoiva alkio
115         int k = 0;
116         for (int j = 0; j < n - 1; ++j) {
117             Rational pi1 = Rational.ZERO;
118             for (int i = j; i < n; ++i) {
119                 Rational pi0 = a[index[i]][j].abs();
120 
121                 // Rivioperaatiomatriisin determinantin seuraaminen
122                 pi0 = pi0.divide(c[index[i]]);
123                 if (pi0.greaterThan(pi1)) {
124                     pi1 = pi0;
125                     k = i;
126                 }
127             }
128 
129             //Järjestä rivit pivotointijärjestyksen mukaan
130             int itmp = index[j];
131             index[j] = index[k];
132             index[k] = itmp;
133             for (int i = j + 1; i < n; ++i) {
134                 Rational pj = a[index[i]][j].divide(a[index[j]][j]);
135 
136 //                Rational pj = Count.divide(a[index[i]][j], a[index[j]][j]);
137                 // Sijoitetaan pivotointisuhteet diagonaalin alle
138                 a[index[i]][j] = pj;
139 
140                 // Säädetään muut alkiot 
141                 for (int l = j + 1; l < n; ++l) {
142                     a[index[i]][l] = Count.difference(a[index[i]][l], Count.product(pj, a[index[j]][l]));
143                 }
144             }
145 
146         }
147     }
