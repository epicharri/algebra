23  /**
24   *
25   * Graafinen käyttöliittymä. Ohjelmaa käytetään tämän käyttöliittymän avulla.

27  public class Gui extends Application {
28  
29      /**
30       * Metodi luo käyttöliittymän komponentit.
31       *
32       * @param stage
33       * @throws Exception
34       */
35      @Override
36      public void start(Stage stage) throws Exception {
37  
38          //ActionEvent event = new ActionEvent();
39          BorderPane layout = new BorderPane();
40  
41          VBox menu = new VBox();
42          menu.setPadding(new Insets(20, 20, 20, 20));
43          menu.setSpacing(10);
44          TextArea input = new TextArea();
45          Label label = new Label("");
46          input.setPromptText("Syötä vektorijono tai yhtälöryhmä.");
47          Tooltip tooltipInput = new Tooltip("Syötä tähän vektorijono tai "
48                  + "yhtälöryhmä.");
49          Tooltip.install(input, tooltipInput);
50  
51          input.prefColumnCountProperty().bind(input.textProperty().length());
52  
53          TextArea answer = new TextArea();
54          answer.setWrapText(true);
55  
56          Button buttonA = new Button("Vektorijonon vapaus");
57          Button buttonB = new Button("Ratkaise yhtälöryhmä");
58          Button buttonC = new Button("Arvo yhtälöryhmä");
59          Button buttonD = new Button("Tyhjennä kentät");
60          Button buttonE = new Button("Käyttöohje");
61          buttonA.setMinWidth(200);
62          buttonB.setMinWidth(200);
63          buttonC.setMinWidth(200);
64          buttonD.setMinWidth(200);
65          buttonE.setMinWidth(200);
66          Tooltip tooltipA = new Tooltip("Ratkaisee onko vektorijono vapaa.");
67          Tooltip.install(buttonA, tooltipA);
68          Tooltip tooltipB = new Tooltip("Ratkaisee kvadraattisen yhtälöryhmän\njos sille löytyy yksikäsitteinen ratkaisu.");
69          Tooltip.install(buttonB, tooltipB);
70          Tooltip tooltipC = new Tooltip("Luo satunnainen yhtälöryhmä");
71          Tooltip.install(buttonC, tooltipC);
72          Tooltip tooltipD = new Tooltip("Tyhjennä kentät");
73          Tooltip.install(buttonD, tooltipD);
74          Tooltip tooltipE = new Tooltip("Antaa ohjeet");
75          Tooltip.install(buttonE, tooltipE);
76  
77          menu.getChildren().addAll(label, input, buttonA, buttonB, buttonC, buttonD, buttonE, answer);
78          layout.setTop(menu);
79  
80          buttonA.setOnAction((ActionEvent eventA) -> {
81              answer.clear();
82              String setOfVectors = input.getText();
83              if (!setOfVectors.trim().isEmpty()) {
84                  answer.setText("Ratkaistaan vektorijonon vapautta.");
85                  try {
86                      String freedom = Solution.solveLinearDependency(setOfVectors);
87                      answer.setText(freedom);
88                  } catch (Exception e) {
89                      answer.setText(e.getMessage());
90                  }
91              }
92          });
93  
94          buttonB.setOnAction((eventB) -> {
95              String system = input.getText();
96              if (!system.trim().isEmpty()) {
97                  try {
98                      String solution = Solution.solveLinearSystem(system);
99                      answer.setText(solution);
100                 } catch (Exception e) {
101                     answer.setText(e.getMessage());
102                 }
103             }
104         });
105 
106         buttonC.setOnAction(new EventHandler<ActionEvent>() {
107             @Override
108             public void handle(ActionEvent eventC) {
109                 answer.clear();
110                 LinearSystem system;
111                 try {
112                     system = RandomLinearSystem.create(10, 10, 9);
113                     input.setText(system.toString());
114                 } catch (Exception ex) {
115                     input.setText(ex.getMessage());
116                 }
117             }
118         });
119 
120         buttonD.setOnAction(new EventHandler<ActionEvent>() {
121             @Override
122             public void handle(ActionEvent eventD) {
123                 answer.clear();
124                 input.clear();
125             }
126         });
127 
128         buttonE.setOnAction(new EventHandler<ActionEvent>() {
129             @Override
130             public void handle(ActionEvent eventE) {
131                 answer.clear();
132                 answer.setText(Instructions.INSTRUCTIONS);
133             }
134         });
135 
136         Scene scene = new Scene(layout, 1000, 600);
137         scene.getStylesheets().add(getClass().getResource("/materialdesign.css").toExternalForm());
138         scene.getStylesheets().add(getClass().getResource("/elementsofharmony.css").toExternalForm());
139         Image heart = new Image(getClass().getResource("/like.png").toExternalForm());
140         stage.getIcons().add(heart);
141         stage.setTitle("LinisLove");
142         stage.setScene(scene);
143         stage.show();
144 
145     }
